let n = 6;

for (let i = 1; i <= n; i++) {
  // printing stars and spaces
  for (let j = 0; j < i; j++) {
    if(i === n) {
      process.stdout.write('*')
    }
    else {
      if (j == 0 || j == i - 1) {
        process.stdout.write('*')
      }
      else {
        process.stdout.write(' ')
      }
    }
  }
  console.log();
}
